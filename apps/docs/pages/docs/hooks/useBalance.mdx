# useBalance

React Hook for fetching balance information for Coins or CW20 tokens.

```ts
import { useBalance } from "@wizard-ui/react";
```

## Usage

```tsx
import { useBalance, useWallet } from "@wizard-ui/react";

function App() {
  const { address } = useWallet();
  const { data, isError, isLoading } = useBalance({ address, token: "uluna" });

  if (isLoading) return <div>Fetching balance…</div>;
  if (isError) return <div>Error fetching balance</div>;
  return <div>Balance: {data}</div>;
}
```

## Return Value

```tsx
{
  data?: string
  error?: Error
  isIdle: boolean
  isLoading: boolean
  isFetching: boolean
  isSuccess: boolean
  isError: boolean
  isFetched: boolean
  isRefetching: boolean
  refetch: (options: {
    throwOnError: boolean
    cancelRefetch: boolean
  }) => Promise<string>
  status: 'idle' | 'error' | 'loading' | 'success'
}
```

## Configuration

### address

Address to fetch balance for.

```tsx {5}
import { useBalance } from "@wizard-ui/react";

function App() {
  const { data, isError, isLoading } = useBalance({
    address: "osmo1vdg4c25t53sqjdfq0tz32g75rlvng5mdz8rk0g",
    token: "uosmo", // native or cw20
  });

  if (isLoading) return <div>Fetching balance…</div>;
  if (isError) return <div>Error fetching balance</div>;
  return <div>Balance: {data}</div>;
}
```

### token

Address for CW20 token or native denom.

```tsx {6}
import { useBalance } from "@wizard-ui/react";

function App() {
  const { data, isError, isLoading } = useBalance({
    address: "osmo1vdg4c25t53sqjdfq0tz32g75rlvng5mdz8rk0g",
    token: "osmo12z0kqd9y28znzjk7pa8e0646nmhrctxnw0nj7265hzgazzml7uuqe88thx", // native or cw20
  });

  if (isLoading) return <div>Fetching balance…</div>;
  if (isError) return <div>Error fetching balance</div>;
  return <div>Balance: {data}</div>;
}
```
